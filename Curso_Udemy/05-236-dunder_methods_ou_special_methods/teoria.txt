# Todas as ações realizas pelo Python, são feitas por métodos especiais. Esses, são built-in
# methods. Ou seja, métodos embutidos do Python.
# Obs.: não é possível criarmos métodos com dunders (underlines duplos: __).

# Abaixo veremos alguns dos métodos especiais mais comuns:
# __lt__(self,other) - self < other
# __le__(self,other) - self <= other
# __gt__(self,other) - self > other
# __ge__(self,other) - self >= other
# __eq__(self,other) - self == other
# __ne__(self,other) - self != other
# __add__(self,other) - self + other
# __sub__(self,other) - self - other
# __mul__(self,other) - self * other
# __truediv__(self,other) - self / other
# __neg__(self) - -self
# __str__(self) - str
# __repr__(self) - str
